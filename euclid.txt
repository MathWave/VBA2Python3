Function NextFirst(first, second)
    If second > first Then
        NextFirst = first
    Else
        NextFirst = first - second
    End If
End Function

Function NextSecond(first, second)
    If first > second Then
        NextSecond = second
    Else
        NextSecond = second - first
    End If
End Function

Function IsPossible(first, second)
    IsPossible = True
End Function

Function GetInfo()
    GetInfo = Array(2, 0)
End Function

Function StopCondition(first, second)
    If first <= 1 Or second <= 1 Then
        return True
    Else
        return False
    End If
End Function

Function Symb(n)
    If n Mod 5 = 0 Then
        Symb = "0"
    ElseIf n Mod 5 = 1 Then
        Symb = "1"
    ElseIf n Mod 5 = 2 Then
        Symb = "2"
    ElseIf n Mod 5 = 3 Then
        Symb = "3"
    Else
        Symb = "4"
End Function

Function GetHyperState(first, second)
    GetHyperState = Symb(first) & "-" & Symb(second)
End Function

Function GetStartPosition()
    GetStartPosition = Array(Rnd(1000000), Rnd(1000000))
End Function
