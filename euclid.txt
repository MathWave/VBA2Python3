Function NextFirst(first, second)
    If second > first Then
        NextFirst = first
    Else
        NextFirst = first - second
    End If
End Function

Function NextSecond(first, second)
    If first > second Then
        NextSecond = second
    Else
        NextSecond = second - first
    End If
End Function

Function IsPossible(first, second)
    IsPossible = True
End Function

Function GetInfo()
    GetInfo = Array(2, 0)
End Function

Function StopCondition(first, second)
    If first = 0 Or second = 0 Then
        return True
    Else
        return False
    End If
End Function

Function Symb(n)
    Symb = CStr(n Mod 5)
End Function

Function GetHyperState(first, second)
    GetHyperState = Symb(first) & "-" & Symb(second)
End Function

Function GetStartPosition()
    GetStartPosition = Array(Rnd(1000), Rnd(1000))
End Function
